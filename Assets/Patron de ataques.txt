public float tiempoEntreAtaques = 5f;
    public GameObject[] ataques;

    private float tiempoTranscurrido;
    private int indiceAtaqueActual;

    void Update()
    {
        tiempoTranscurrido += Time.deltaTime;

        if (tiempoTranscurrido >= tiempoEntreAtaques)
        {
            tiempoTranscurrido = 0f;
            EjecutarAtaque();
        }
    }

    void EjecutarAtaque()
    {
        if (ataques.Length == 0)
        {
            Debug.LogWarning("No se han asignado ataques al jefe.");
            return;
        }

        // Seleccionar un ataque aleatorio
        indiceAtaqueActual = Random.Range(0, ataques.Length);

        // Ejecutar el ataque seleccionado
        Instantiate(ataques[indiceAtaqueActual], transform.position, Quaternion.identity);
    }
